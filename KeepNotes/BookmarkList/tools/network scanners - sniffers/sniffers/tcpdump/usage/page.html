tcpdump -r file.pcap #read a pcap file, can pipe this into other tools for processing
tcpdump src host <ip> -r file.pcap #use tcpdump built in filters on pcap files
tcpdump -X -r file.pcap #print full Hex and ASCII output from pcap
tcpdump -A 'tcp[13] = 24' -r file.pcap #filter for specific packet flags, ACK and PSH. Filter on 14th byte of tcp[]/



tcpdump filter on tcp[tcpflags] == tcp-syn #see when something first initiates a TCP connection to us
tcpdump filter on icmp #see when something pings us
